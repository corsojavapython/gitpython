CREATE TABLE IF NOT EXISTS `c3db`.`PRODOTTI` (
 `CODICE` INT AUTO_INCREMENT NOT NULL,
  `CATEGORIA` VARCHAR(45) NOT NULL,
  `NOME PRODOTTO` VARCHAR(45) NOT NULL,
  `VENDITORE_CODICE` VARCHAR(32) NOT NULL,
  `DESCRIZIONE` TEXT NOT NULL,
  `PREZZO` DECIMAL(10, 2) NOT NULL,
  `FOTO` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`CODICE`),
  FOREIGN KEY (`VENDITORE_CODICE`) REFERENCES `UTENTE` (`CODICE`)
);


CREATE TABLE IF NOT EXISTS `c3db` . `WALLET` (
  `ID` INT AUTO_INCREMENT PRIMARY KEY,
  `CODICE_VENDITORE` VARCHAR(32) NOT NULL,
  `SALDO` DECIMAL(10, 2) NOT NULL,
  
  FOREIGN KEY (`CODICE_VENDITORE`) REFERENCES `UTENTE`(`CODICE`)
);



REATE TABLE IF NOT EXISTS c3db.TRANSAZIONI (
  ID INT AUTO_INCREMENT PRIMARY KEY,
  CODICE_ACQUIRENTE VARCHAR(32) NOT NULL,
  ID_PRODOTTO INT NOT NULL,
  IMPORTO_PRODOTTO DECIMAL(10, 2) NOT NULL,
  DATA_TRANSAZIONE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  DATA_PAGAMENTO TIMESTAMP DEFAULT NULL,

  FOREIGN KEY (CODICE_ACQUIRENTE) REFERENCES UTENTE(CODICE),
  FOREIGN KEY (ID_PRODOTTO) REFERENCES PRODOTTI(CODICE)
);
